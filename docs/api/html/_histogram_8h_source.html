44 template <typename T>
45 inline int search_bin_idx(T value, T *bin_edges, int n_bins){
46  int b, e, m;
47 
48  // ignore values outside of the bounds
49  if ((value < bin_edges[0]) || (value > bin_edges[n_bins - 2])) {
50  return -1;
51  }
52 
53  b = 0;
54  e = n_bins;
55  do {
56  m = (e - b) / 2 + b;
57  if (value > bin_edges[m]) {
58  b = m;
59  } else {
60  e = m;
61  }
62  } while ((value < bin_edges[m]) || (value >= bin_edges[m + 1]));
63  return m;
64 }

229  void update(){
230  int axis_index;
231  int global_bin_idx;
232  int global_bin_offset;
233  int n_axis;
234  int current_bin_idx, current_n_bins;
235  Axis<T> *current_axis;
236  T data_value;
237 
238  // update the axes
239  for(const auto& p : axes){
240  axes[p.first].update();
241  }
242 
243  // initialize a new empty histogram
244  // clear the memory of the old histogram
245  free(histogram);
246  // 1. count the total number of bins
247  n_total_bins = 1;
248  n_axis = 0;
249  for(const auto& p : axes){
250  axis_index = p.first;
251  n_axis += 1;
252  n_total_bins *= axes[axis_index].getNumberOfBins();
253  }
254  // 2. fill the histogram with zeros
255  histogram = (T*) malloc(sizeof(T) * (n_total_bins));
256  for(global_bin_idx=0; global_bin_idx < n_total_bins; global_bin_idx++){
257  histogram[global_bin_idx] = 0.0;
258  }
259 
260  // Fill the histogram
261  // Very instructive for multi-dimensional array indexing
262  // https://eli.thegreenplace.net/2015/memory-layout-of-multi-dimensional-arrays/
263  bool is_inside;
264  for(int i_row=0; i_row<n_rows_data; i_row++){
265  // in this loop the position within the 1D array is calculated
266  global_bin_offset = 0;
267  is_inside = true;
268  for(const auto& p : axes){
269  axis_index = p.first;
270  current_axis = &axes[axis_index];
271 
272  data_value = data[i_row*n_axis + axis_index];
273  current_bin_idx = current_axis->getBinIdx(data_value);
274  current_n_bins = current_axis->getNumberOfBins();
275 
276  if( (current_bin_idx < 0) || (current_bin_idx >= current_n_bins) ){
277  is_inside = false;
278  break;
279  }
280  global_bin_offset = current_bin_idx + current_n_bins * global_bin_offset;
281  }
282  if(is_inside){
283  histogram[global_bin_offset] += 1.0;
284  }
285  }
286  }

341 void histogram1D(
342  T *data, int n_data,
343  double *weights, int n_weights,
344  T *bin_edges, int n_bins,
345  double *hist, int n_hist,
346  const char *axis_type,
347  bool use_weights
348 ) {
349  T v; // stores the data value in iterations
350  int i, bin_idx;
351  T lower, upper, bin_width;
352  bool is_log10 = !strcmp(axis_type, "log10");
353  bool is_lin = !strcmp(axis_type, "lin");
354 
355  if (is_lin || is_log10) {
356  if (is_log10) {
357  lower = std::log10(bin_edges[0]);
358  upper = std::log10(bin_edges[n_bins - 1]);
359  } else {
360  lower = bin_edges[0];
361  upper = bin_edges[n_bins - 1];
362  }
363  bin_width = (upper - lower) / (n_bins - 1);
364 
365  for (i = 0; i < n_data; i++) {
366  v = data[i];
367  if(is_log10){
368  if(v == 0){
369  continue;
370  } else {
371  v = std::log10(v);
372  }
373  }
374  bin_idx = calc_bin_idx(lower, bin_width, v);
375  // ignore values outside of the bounds
376  if ((bin_idx <= n_bins) && (bin_idx >= 0)){
377  hist[bin_idx] += (use_weights) ? weights[i] : 1;
378  }
379  }
380  } else {
381  for (i = 0; i < n_data; i++) {
382  v = data[i];
383  bin_idx = search_bin_idx(v, bin_edges, n_bins);
384  if(bin_idx > 0)
385  hist[bin_idx] += (use_weights) ? weights[i] : 1;
386  }
387  }
388 }
